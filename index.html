const express = require('express'); const app = express(); app.use(express.json()); /* In-memory data */ let courses = [ { id: 1, title: "Web Development", description: "HTML, CSS, JS", videos: [] }, { id: 2, title: "Data Science", description: "Python & ML", videos: [] } ]; let enrollments = []; let progress = []; /* API ROUTES */ // Get courses app.get('/courses', (req, res) => res.json(courses)); // Enroll app.post('/enroll', (req, res) => { enrollments.push(req.body); res.json({ message: "Enrolled" }); }); // Progress app.post('/progress', (req, res) => { progress.push(req.body); res.json({ message: "Progress saved" }); }); // Certificate app.get('/certificate', (req, res) => { const { name, courseId } = req.query; res.send(`🎓 Certificate: ${name} completed course ${courseId}`); }); // Add course app.post('/courses', (req, res) => { const c = req.body; c.id = courses.length + 1; c.videos = []; courses.push(c); res.json(c); }); // Delete course app.delete('/courses/:id', (req, res) => { courses = courses.filter(c => c.id != req.params.id); res.json({ message: "Deleted" }); }); // Add video app.post('/courses/:id/videos', (req, res) => { const course = courses.find(c => c.id == req.params.id); if (course) { course.videos.push(req.body.url); res.json({ message: "Video added" }); } }); /* FRONTEND (HTML INSIDE NODE) */ app.get('/', (req, res) => { res.send(` EduLearn

`); }); app.listen(3000, () => console.log("http://localhost:3000"));
